### Remark
- 先把尾巴對齊後再來判斷有沒有一樣
	- 要先找長度
	- 把長的前面先移除掉
- linked list不能用 headA, headB = headB, headA !!!!!!
---
### Code
```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def getIntersectionNode(self, headA: ListNode, headB: ListNode) -> Optional[ListNode]:
        def getLength(head):
            length = 0
            while head:
                head = head.next
                length += 1
            return length
        
        def moveForward(head, length):
            for i in range(length):
                head = head.next
            return head
        
        lenA, lenB = getLength(headA), getLength(headB)
        if lenA > lenB:
            headA = moveForward(headA, lenA-lenB)
        else:
            headB = moveForward(headB, lenB-lenA)
        while headA:
            if headA == headB:
                return headA
            headA = headA.next
            headB = headB.next
        return None
```
